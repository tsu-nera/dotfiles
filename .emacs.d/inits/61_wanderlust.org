* HTMLメール対策
emacs-w3mをつかう。w3mが必要。(apt-get install w3m)

#+begin_src emacs-lisp
(when linux-p
(require 'wl)

;; HTMLファイルは表示しない。
(setq mime-setup-enable-inline-html nil)

;; HTMLメールはw3mで
(require 'mime-setup)
(when (locate-library "mime-w3m")
  (require 'w3m-ems)
  (require 'w3m-load)
  (require 'mime-w3m)
  (setq mime-w3m-safe-url-regexp nil)
  )
)
#+end_src
